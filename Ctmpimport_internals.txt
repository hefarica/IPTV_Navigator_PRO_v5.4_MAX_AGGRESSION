Private Sub FE_SyncServerJS(nm As String, u As String, usr As String, pw As String)
    Dim ws As Worksheet: Set ws = ThisWorkbook.Sheets(SHEET_SERVERS)
    If u = "" Then Exit Sub
    Dim r As Long, lr As Long, found As Boolean
    lr = ws.Cells(ws.rows.Count, 3).End(xlUp).row
    For r = 4 To lr
        If CStr(ws.Cells(r, 3).value) = u Then
            ws.Cells(r, 1).value = "X"
            ws.Cells(r, 4).value = usr
            ws.Cells(r, 5).value = pw
            If nm <> "" Then ws.Cells(r, 9).value = nm & " (from frontend)"
            found = True: Exit For
        End If
    Next r
    If Not found Then
        r = lr + 1
        ws.Cells(r, 1).value = "X"
        ws.Cells(r, 2).value = r - 3
        ws.Cells(r, 3).value = u
        ws.Cells(r, 4).value = usr
        ws.Cells(r, 5).value = pw
        ws.Cells(r, 6).value = 1
        ws.Cells(r, 8).value = "Active"
        ws.Cells(r, 9).value = nm & " (from frontend)"
    End If
End Sub

Private Sub FE_UpdateProfileCell(ws As Worksheet, pid As String, attr As String, newVal As String)
    If newVal = "" Then Exit Sub
    Dim col As Integer
    Select Case UCase(pid)
        Case "P0": col = 2
        Case "P1": col = 3
        Case "P2": col = 4
        Case "P3": col = 5
        Case "P4": col = 6
        Case "P5": col = 7
        Case Else: Exit Sub
    End Select
    Dim r As Long, lr As Long
    lr = ws.Cells(ws.rows.Count, 1).End(xlUp).row
    For r = 3 To lr
        If InStr(LCase(CStr(ws.Cells(r, 1).value)), LCase(attr)) > 0 Then
            ws.Cells(r, col).value = newVal
            ws.Cells(r, col).Interior.Color = RGB(226, 239, 218)
            Exit Sub
        End If
    Next r
End Sub
