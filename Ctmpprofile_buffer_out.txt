Public Function ConfigVal(paramName As String) As String
    Dim ws As Worksheet, r As Long, lastRow As Long
    Set ws = ThisWorkbook.Sheets(SHEET_CONFIG)
    lastRow = ws.Cells(ws.Rows.Count, 2).End(xlUp).row
    For r = 3 To lastRow
        If LCase(CStr(ws.Cells(r, 2).value)) = LCase(paramName) Then
            ConfigVal = CStr(ws.Cells(r, 3).value): Exit Function
        End If
    Next r
    ConfigVal = ""
End Function

Public Function ProfileVal(profileId As String, attrName As String) As String
    Dim ws As Worksheet, r As Long, lastRow As Long, colIdx As Integer
    Set ws = ThisWorkbook.Sheets(SHEET_PROFILES)
    Select Case UCase(profileId)
        Case "P0": colIdx = 2
        Case "P1": colIdx = 3
        Case "P2": colIdx = 4
        Case "P3": colIdx = 5
        Case "P4": colIdx = 6
        Case "P5": colIdx = 7
        Case Else: colIdx = 2
    End Select
    lastRow = ws.Cells(ws.Rows.Count, 1).End(xlUp).row
    For r = 3 To lastRow
        If LCase(CStr(ws.Cells(r, 1).value)) = LCase(attrName) Or _
           InStr(LCase(CStr(ws.Cells(r, 1).value)), LCase(attrName)) > 0 Then
            ProfileVal = CStr(ws.Cells(r, colIdx).value): Exit Function
        End If
    Next r
    ProfileVal = ""
End Function
