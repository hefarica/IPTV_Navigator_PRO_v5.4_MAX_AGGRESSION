549: * Extrae toda la información estructural de la lista: entradas EXTINF,
550: * directivas de jugador, cabeceras HTTP, perfiles APE, y metadatos del stream.
551: * No realiza ninguna petición de red ni modifica el contenido.
552: */
553: class ManifestAnatomyParser
554: {
555: /**
556: * Analiza el contenido bruto de una lista M3U y devuelve su ADN completo.
557: *
558: * @param  string $raw Contenido bruto del archivo M3U
559: * @return array  Estructura manifest_dna completa
560: */
561: public function parse(string $raw): array
562: {
563: // Dividir en líneas limpias
564: $lines = array_values(array_filter(
565: array_map('trim', explode("\n", str_replace("\r\n", "\n", $raw))),
566: static fn(string $line): bool => $line !== ''
567: ));
568: 
569: // Clasificar líneas en tags y streams
570: $classified = $this->classifyLines(lines: $lines);
571: 
572: // Analizar cada entrada EXTINF
573: $extinfEntries = [];
574: foreach ($classified['tags'] as $idx => $tagLine) {
575: if (str_starts_with($tagLine, '#EXTINF:')) {
576: $parsed = $this->parseExtInf(line: $tagLine);

583: $dna = [
584: 'raw_lines'        => $lines,

586: 'streams'          => $classified['streams'],
587: 'extinf_entries'   => $extinfEntries,
588: 'extvlcopt'        => $this->extractDirectivesByPrefix(tags: $classified['tags'], prefix: '#EXTVLCOPT:'),
589: 'kodiprop'         => $this->extractDirectivesByPrefix(tags: $classified['tags'], prefix: '#KODIPROP:'),

591: 'extx_tags'        => $this->extractDirectivesByPrefix(tags: $classified['tags'], prefix: '#EXT-X-'),
592: 'extattrfromurl'   => $this->extractDirectivesByPrefix(tags: $classified['tags'], prefix: '#EXTATTRFROMURL:'),

598: 'streams' => $classified['streams'],

647: /**
648: * Parsea una línea #EXTINF y extrae todos los atributos y el título.
649: *
650: * Formato: #EXTINF:duration key="value" ...,Title
651: *
652: * @param  string $line Línea EXTINF completa
653: * @return array{duration: int|string, attributes: array<string, string>, title: string}
654: */
655: private function parseExtInf(string $line): array
656: {
657: // Eliminar el prefijo #EXTINF:
658: $content = substr($line, 8);
659: $content = ltrim($content);
660: 
661: // Separar la parte de atributos/título de la duración
662: $commaPos = strpos($content, ',');
663: if ($commaPos === false) {
664: return ['duration' => trim($content), 'attributes' => [], 'title' => trim($content)];
665: }
666: 
667: $duration = trim(substr($content, 0, $commaPos));
668: $afterComma = substr($content, $commaPos + 1);
669: $title = trim($afterComma);
670: 
671: // La parte antes de la coma puede contener la duración y atributos
672: $beforeComma = substr($content, 0, $commaPos);
673: 
674: // Extraer atributos tipo key="value" y key=value
675: $attributes = [];
676: 
677: // Patrón para atributos con comillas: key="value" o key='value'
678: if (preg_match_all(
679: '/([a-zA-Z0-9_\-]+)\s*=\s*"([^"]*)"/',
680: $beforeComma,
681: $matches,
682: PREG_SET_ORDER
683: )) {
684: foreach ($matches as $match) {
685: $attributes[$match[1]] = $match[2];
686: }
687: }
688: 
689: // También buscar atributos sin comillas: key=value (solo si no fue capturado con comillas)
690: if (preg_match_all(
691: '/([a-zA-Z0-9_\-]+)\s*=\s*([^\s"]+)/',
692: $beforeComma,
693: $matches,
694: PREG_SET_ORDER
695: )) {
696: foreach ($matches as $match) {
697: if (!isset($attributes[$match[1]])) {
698: $attributes[$match[1]] = $match[2];
699: }
700: }
701: }
702: 
703: // Intentar extraer atributos del título también (algunas listas los ponen ahí)
704: if (preg_match_all(
705: '/([a-zA-Z0-9_\-]+)\s*=\s*"([^"]*)"/',
706: $afterComma,
707: $matches,
708: PREG_SET_ORDER
709: )) {
710: foreach ($matches as $match) {
711: $attributes[$match[1]] = $match[2];
712: // Limpiar el atributo del título
713: $title = str_replace($match[0], '', $title);
714: }
715: }
716: 
717: $title = trim($title, " \t\n\r\0\x0B,");
718: 
719: return [
720: 'duration'   => $duration,
721: 'attributes' => $attributes,
722: 'title'      => $title,
723: ];
724: }

1074: /**
1075: * Motor de normalización idempotente.
1076: *
1077: * Deduplica directivas, resuelve conflictos y hereda políticas del ADN.
1078: * Garantiza que la misma entrada produzca siempre la misma salida.
1079: */
1080: class IdempotentNormalizationEngine
1081: {
1082: /**
1083: * Normaliza todas las directivas y políticas heredadas del ADN.
1084: *
1085: * @param  array $dna ADN del manifiesto (salida de ManifestAnatomyParser)
1086: * @return array{inherited_policies: array, normalized_directives: array}
1087: */
1088: public function normalize(array $dna): array
1089: {
1090: $inheritedPolicies = $this->mergeInheritedPolicies(dna: $dna);
1091: $normalizedDirectives = $this->buildNormalizedDirectives(dna: $dna);
1092: 
1093: return [
1094: 'inherited_policies'     => $inheritedPolicies,
1095: 'normalized_directives'  => $normalizedDirectives,
1096: ];
1097: }
1098: 
1099: /**
1100: * Deduplica directivas por tipo, manteniendo el orden de inserción.
1101: *
1102: * @param  array  $directives Lista de directivas crudas
1103: * @param  string $type       Tipo: 'vlcopt', 'kodiprop', 'exthttp'
1104: * @return array  Directivas deduplicadas
1105: */
1106: private function deduplicateDirectives(array $directives, string $type): array
1107: {
1108: $seen = [];
1109: $result = [];
1110: 
1111: foreach ($directives as $directive) {
1112: $key = match ($type) {
1113: 'vlcopt', 'kodiprop' => strstr($directive, '=', true) ?: $directive,
1114: 'exthttp' => is_array($directive) ? array_key_first($directive) : strstr((string) $directive, '=', true),
1115: default => $directive,
1116: };
1117: 
1118: if ($key !== null && !isset($seen[$key])) {
1119: $seen[$key] = true;
1120: $result[] = $directive;
1121: }
1122: }
1123: 
1124: return $result;
1125: }
1126: 
1127: /**
1128: * Resuelve conflictos de buffer: el valor MÁXIMO gana.
1129: *
1130: * @param  array<int> $values Valores de buffer encontrados (ms)
1131: * @return int|null   Valor máximo o null si no hay valores
1132: */
1133: private function resolveBufferConflicts(array $values): ?int
1134: {
1135: $valid = array_filter(
1136: $values,
1137: static fn($v): bool => is_numeric($v) && $v > 0
1138: );
1139: 
1140: if (empty($valid)) {
1141: return null;
1142: }
1143: 
1144: return (int) max($valid);
1145: }
1146: 
1147: /**
1148: * Resuelve conflictos de codec: el más específico gana.
1149: * Orden de especificidad: av1 > hevc > h264 > vp9 > any
1150: *
1151: * @param  array<string> $values Codecs encontrados
1152: * @return string|null    Codec más específico
1153: */
1154: private function resolveCodecConflicts(array $values): ?string
1155: {
1156: if (empty($values)) {
1157: return null;
1158: }
1159: 
1160: $specificity = ['av1' => 5, 'hevc' => 4, 'h265' => 4, 'vp9' => 3, 'h264' => 2, 'avc' => 2, 'any' => 0];
1161: 
1162: $best = null;
1163: $bestScore = -1;
1164: 
1165: foreach ($values as $codec) {
1166: $low = strtolower(trim($codec));
1167: $score = $specificity[$low] ?? 1;
1168: 
1169: if ($score > $bestScore) {
1170: $bestScore = $score;
1171: $best = $low;
1172: }
1173: }
1174: 
1175: return $best;
1176: }

1178: /**
1179: * Fusiona políticas heredadas del ADN del manifiesto.
1180: * Extrae valores de buffer, codec, HDR, resolución, etc.
1181: */
1182: private function mergeInheritedPolicies(array $dna): array
1183: {
1184: $policies = [
1185: 'buffer'            => null,
1186: 'codec'             => null,
1187: 'hdr'               => [],
1188: 'resolution'        => null,
1189: 'bitrate'           => null,
1190: 'fallback_chain'    => $dna['fallback_chains'],
1191: 'network'           => [],
1192: 'deinterlace'       => null,
1193: 'audio_passthrough' => null,
1194: 'color_depth'       => null,
1195: ];
1196: 
1197: // --- Extraer valores de buffer de EXTVLCOPT ---
1198: $bufferValues = [];
1199: foreach ($dna['extvlcopt'] as $opt) {
1200: if (str_contains(strtolower($opt), 'caching=')) {
1201: $eqPos = strpos($opt, '=');
1202: if ($eqPos !== false) {
1203: $val = (int) substr($opt, $eqPos + 1);
1204: if ($val > 0) {
1205: $bufferValues[] = $val;
1206: }
1207: }
1208: }
1209: }
1210: $policies['buffer'] = $this->resolveBufferConflicts(values: $bufferValues);
1211: 
1212: // --- Extraer codec de hints ---
1213: $codecs = [];
1214: if (!empty($dna['codec_hints']['codec'])) {
1215: $codecs[] = $dna['codec_hints']['codec'];
1216: }
1217: if (isset($dna['ape_attributes']['ape-codec'])) {
1218: $codecs[] = $dna['ape_attributes']['ape-codec'];

1647:

1685: if (preg_match('/group|categoria|category/i', $text)) {

1749: * - Player: 0-33 basado en codec support, HDR, max resolution
1750: * - TV: 0-33 basado en HDR, resolution, nits, color depth
1751: * - Network: 0-33 basado en throughput, latency

1756: {
1757: $player = $capabilities['player'];
1758: $tv = $capabilities['tv'];
1759: $network = $capabilities['network'];
1760: 
1761: // === Puntuación del reproductor (0-33) ===
1762: $playerScore = 0;
1763: 
1764: // Soporte de codecs (0-10)
1765: $codecCount = count($player['codec_support']);
1766: $playerScore += min(10, $codecCount * 3);
1767: 
1768: // Soporte de HDR (0-8)
1769: $playerScore += min(8, count($player['hdr_support']) * 3);
1770: 
1771: // Resolución máxima (0-8)
