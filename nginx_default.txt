server {
    listen 443 ssl default_server;
    listen [::]:443 ssl default_server;
    server_name iptv-ape.duckdns.org 178.156.147.234;
    # SSL Configuration
    ssl_certificate /etc/letsencrypt/live/iptv-ape.duckdns.org/fullchain.pem;
    ssl_certificate_key /etc/letsencrypt/live/iptv-ape.duckdns.org/privkey.pem;
    ssl_protocols TLSv1.2 TLSv1.3;
    # VPS SOLO TEXTO: root donde viven las playlists y scripts PHP
    root /var/www/html;
    index index.php index.html;
    # Logs dedicados
    access_log /var/log/nginx/playlist_access.log;
    error_log /var/log/nginx/playlist_error.log warn;
    server_tokens off;
    # Performance para texto grande
    sendfile on;
    tcp_nopush on;
    tcp_nodelay on;
    # Timeouts defensivos para uploads grandes
    client_max_body_size 600M;
    client_body_timeout 600s;
    send_timeout 600s;
    keepalive_timeout 120s;
    proxy_connect_timeout 60s;
    proxy_send_timeout 600s;
    proxy_read_timeout 600s;

    # Healthcheck
    location = /health {
        add_header Content-Type text/plain;
        return 200 "OK\n";
    }

    # ==============================
    #  LISTS (Proxy al Guardian Engine con toggle)
    #  NOTA: ^~ da prioridad sobre regex patterns
    # ==============================
    location ^~ /lists/ {
        # Proxy al Guardian Engine (puerto 8080)
        # El Guardian Engine decidirá si procesar o servir directamente según el toggle
        proxy_pass http://127.0.0.1:8080/;
        proxy_http_version 1.1;
        proxy_set_header Host $host;
        proxy_set_header X-Real-IP $remote_addr;
        proxy_set_header X-Forwarded-For $proxy_add_x_forwarded_for;
        proxy_set_header X-Forwarded-Proto $scheme;
        
        # Buffers grandes para manejar 170+ headers ABR del Guardian Engine
        proxy_buffer_size 16k;
        proxy_buffers 8 16k;
        proxy_busy_buffers_size 32k;
        
        # Desactivar buffering para streaming
        proxy_buffering off;
        proxy_request_buffering off;
        
        # CORS completo (después del proxy_pass)
        add_header Access-Control-Allow-Origin "*" always;
        add_header Access-Control-Allow-Methods "GET, HEAD, OPTIONS" always;
        add_header Access-Control-Allow-Headers "Range, DNT, User-Agent, X-Requested-With, If-Modified-Since, Cache-Control, Content-Type" always;
        add_header Access-Control-Expose-Headers "*" always;
    }

    # ==============================
    #  APE GUARDIAN ENGINE (Python - Puerto 8080)
    #  NOTA: ^~ da prioridad sobre regex patterns
    # ==============================
    location ^~ /guardian/ {
        # CORS completo
        add_header Access-Control-Allow-Origin "*" always;
        add_header Access-Control-Allow-Methods "GET, HEAD, OPTIONS" always;
        add_header Access-Control-Allow-Headers "Range, DNT, User-Agent, X-Requested-With, If-Modified-Since, Cache-Control, Content-Type" always;
        add_header Access-Control-Expose-Headers "*" always;

        # Preflight OPTIONS
        if ($request_method = OPTIONS) {
            add_header Access-Control-Allow-Origin "*" always;
            add_header Access-Control-Allow-Methods "GET, HEAD, OPTIONS" always;
            add_header Access-Control-Allow-Headers "Range, DNT, User-Agent, X-Requested-With, If-Modified-Since, Cache-Control, Content-Type" always;
            add_header Access-Control-Max-Age 86400 always;
            add_header Content-Type text/plain;
            add_header Content-Length 0;
            return 204;
        }

        # Rewrite para eliminar /guardian/ del path
        rewrite ^/guardian/(.*)$ /$1 break;

        # Proxy al Guardian Engine v17 ULTRA-AGGRESSIVE (puerto 8081)
        proxy_hide_header Access-Control-Allow-Origin;
        proxy_hide_header Access-Control-Expose-Headers;
        proxy_pass http://127.0.0.1:8081;
        proxy_http_version 1.1;
        proxy_set_header Host $host;
        proxy_set_header X-Real-IP $remote_addr;
        proxy_set_header X-Forwarded-For $proxy_add_x_forwarded_for;
        proxy_set_header X-Forwarded-Proto $scheme;
        proxy_buffering off;
        proxy_request_buffering off;
    }

    # ==============================
    #  UPLOAD SERVER (Node.js - Puerto 9000)
    # ==============================
    location /upload/ {
        # CORS completo para uploads
        add_header Access-Control-Allow-Origin "*" always;
        add_header Access-Control-Allow-Methods "GET, POST, OPTIONS" always;
        add_header Access-Control-Allow-Headers "Content-Type, Authorization" always;
        add_header Access-Control-Max-Age 86400 always;

        # Preflight OPTIONS
        if ($request_method = OPTIONS) {
            add_header Access-Control-Allow-Origin "*" always;
            add_header Access-Control-Allow-Methods "GET, POST, OPTIONS" always;
            add_header Access-Control-Allow-Headers "Content-Type, Authorization" always;
            add_header Access-Control-Max-Age 86400 always;
            add_header Content-Type text/plain;
            add_header Content-Length 0;
            return 204;
        }

        # Proxy al Upload Server
        proxy_pass http://127.0.0.1:9000;
        proxy_http_version 1.1;
        proxy_set_header Host $host;
        proxy_set_header X-Real-IP $remote_addr;
        proxy_set_header X-Forwarded-For $proxy_add_x_forwarded_for;
        proxy_set_header X-Forwarded-Proto $scheme;
        proxy_request_buffering off;
        proxy_buffering off;
    }

    # ==============================
    #  UPLOAD VERIFY (Node.js - Puerto 9000)
    # ==============================
    location = /upload-verify {
        # CORS completo
        add_header Access-Control-Allow-Origin "*" always;
        add_header Access-Control-Allow-Methods "GET, POST, OPTIONS" always;
        add_header Access-Control-Allow-Headers "Content-Type" always;

        if ($request_method = OPTIONS) {
            add_header Access-Control-Allow-Origin "*" always;
            add_header Access-Control-Allow-Methods "GET, POST, OPTIONS" always;
            add_header Access-Control-Allow-Headers "Content-Type" always;
            add_header Content-Type text/plain;
            add_header Content-Length 0;
            return 204;
        }

        proxy_pass http://127.0.0.1:9000;
        proxy_http_version 1.1;
        proxy_set_header Host $host;
        proxy_set_header X-Real-IP $remote_addr;
    }

    # ==============================
    #  PROXY DINÁMICO (Gunicorn - Puerto 8080)
    # ==============================
    location ^~ /proxy/ {
        rewrite ^/proxy/(.*)$ /$1 break;
        # CORS completo
        add_header Access-Control-Allow-Origin "*" always;
        add_header Access-Control-Allow-Methods "GET, OPTIONS" always;
        add_header Access-Control-Allow-Headers "Range, DNT, User-Agent, X-Requested-With" always;
        add_header Access-Control-Expose-Headers "Content-Length, Content-Range" always;

        if ($request_method = OPTIONS) {
            add_header Access-Control-Allow-Origin "*" always;
            add_header Access-Control-Allow-Methods "GET, OPTIONS" always;
            add_header Access-Control-Allow-Headers "Range, DNT, User-Agent, X-Requested-With" always;
            add_header Content-Type text/plain;
            add_header Content-Length 0;
            return 204;
        }

        # Proxy al Headers Proxy Server
        proxy_hide_header Access-Control-Allow-Origin;
        proxy_hide_header Access-Control-Expose-Headers;
        proxy_pass http://127.0.0.1:8080;
        proxy_http_version 1.1;
        proxy_set_header Host $host;
        proxy_set_header X-Real-IP $remote_addr;
        proxy_set_header X-Forwarded-For $proxy_add_x_forwarded_for;
        proxy_set_header X-Forwarded-Proto $scheme;
        proxy_buffering off;
    }

    # ==============================
    #  API SERVER (Gunicorn - Puerto 5001)
    # ==============================
    
    location = /api/health {
        add_header Access-Control-Allow-Origin "*" always;
        add_header Access-Control-Allow-Methods "GET, OPTIONS" always;
        add_header Content-Type "application/json" always;
        if ($request_method = OPTIONS) { return 200; }
        return 200 '{"status":"ok","timestamp":1700000000}';
    }

    location = /api/jwt-config {
        add_header Access-Control-Allow-Origin "*" always;
        add_header Access-Control-Allow-Methods "GET, OPTIONS" always;
        add_header Content-Type "application/json" always;
        if ($request_method = OPTIONS) { return 200; }
        return 200 '{"jwt_duration":8760,"active":true}';
    }

    location /api/ {
        # CORS completo para API
        add_header Access-Control-Allow-Origin "*" always;
        add_header Access-Control-Allow-Methods "GET, POST, PUT, DELETE, OPTIONS" always;
        add_header Access-Control-Allow-Headers "Content-Type, Authorization, X-Upload-Id, X-Chunk-Index, X-Total-Chunks, X-Chunk-SHA256, X-Chunk-MD5, X-Secret-Key" always;
        add_header Access-Control-Expose-Headers "Content-Length, Content-Range" always;
        add_header Access-Control-Max-Age 86400 always;

        # Preflight OPTIONS
        if ($request_method = OPTIONS) {
            add_header Access-Control-Allow-Origin "*" always;
            add_header Access-Control-Allow-Methods "GET, POST, PUT, DELETE, OPTIONS" always;
            add_header Access-Control-Allow-Headers "Content-Type, Authorization, X-Upload-Id, X-Chunk-Index, X-Total-Chunks, X-Chunk-SHA256, X-Chunk-MD5, X-Secret-Key" always;
            add_header Access-Control-Max-Age 86400 always;
            add_header Content-Type text/plain;
            add_header Content-Length 0;
            return 204;
        }

        # Proxy al APE API Server
        proxy_pass http://127.0.0.1:5001;
        proxy_http_version 1.1;
        proxy_set_header Host $host;
        proxy_set_header X-Real-IP $remote_addr;
        proxy_set_header X-Forwarded-For $proxy_add_x_forwarded_for;
        proxy_set_header X-Forwarded-Proto $scheme;
        proxy_request_buffering off;
        proxy_buffering off;
    }

    # ==============================
    #  ATOMIC DEPLOYMENT (ZIP UPLOAD)
    # ==============================
    location = /upload.php {
        if ($request_method = OPTIONS) {
            add_header Access-Control-Allow-Origin "*" always;
            add_header Access-Control-Allow-Methods "GET, HEAD, POST, OPTIONS" always;
            add_header Access-Control-Allow-Headers "*" always;
            add_header Content-Length 0 always;
            return 204;
        }
        
        # Extrema tolerancia para subidas atómicas masivas
        client_max_body_size 5000M;
        client_body_timeout 3600s;
        fastcgi_read_timeout 3600s;
        send_timeout 3600s;
        
        # CORS headers are sent by PHP itself — do NOT add them here to avoid duplication
        include snippets/fastcgi-php.conf;
        fastcgi_pass unix:/run/php/php8.3-fpm.sock;
    }

    # ==============================
    #  PHP UPLOAD SCRIPTS (CHUNKED)
    # ==============================
    # Preflight OPTIONS para PHP — SEPARADO del bloque FastCGI para evitar el bug "if is evil"
    location ~ \.php$ {
        # PHP-FPM (sin if/return dentro del contexto FastCGI)
        include snippets/fastcgi-php.conf;
        fastcgi_pass unix:/run/php/php8.3-fpm.sock;
        fastcgi_read_timeout 300s;
        fastcgi_send_timeout 300s;
    }
    # ==============================
    #  PLAYLIST M3U8 (SOLO TEXTO)
    # ==============================
    location ~* \.m3u8$ {
        # MIME correcto
        default_type application/vnd.apple.mpegurl;
        types { application/vnd.apple.mpegurl m3u8; }
        # CORS completo
        add_header Access-Control-Allow-Origin "*" always;
        add_header Access-Control-Allow-Methods "GET, HEAD, OPTIONS" always;
        add_header Access-Control-Allow-Headers "Range, DNT, User-Agent, X-Requested-With, If-Modified-Since, Cache-Control, Content-Type" always;
        add_header Access-Control-Expose-Headers "Content-Length, Content-Range, ETag, Last-Modified" always;
        # Preflight (sin redirects; 200 directo)
        if ($request_method = OPTIONS) {
            add_header Content-Type text/plain;
            add_header Content-Length 0;
            return 200;
        }
        # Range requests
        add_header Accept-Ranges bytes always;
        # Cache control
        add_header Cache-Control "public, max-age=300" always;
        expires 5m;
        # NO REDIRECTS EN PLAYLIST
        limit_except GET HEAD {
            deny all;
        }
    }
    # ==============================
    #  STATIC FILES (icons, etc)
    # ==============================
    location / {
        # CORS for other requests
        add_header Access-Control-Allow-Origin "*" always;
        add_header Access-Control-Allow-Methods "GET, POST, OPTIONS" always;
        
        if ($request_method = OPTIONS) {
            add_header Access-Control-Allow-Origin "*" always;
            add_header Access-Control-Allow-Methods "GET, POST, OPTIONS" always;
            add_header Content-Type text/plain;
            add_header Content-Length 0;
            return 204;
        }
        
        try_files $uri $uri/ =404;
    }
    # ==============================
    #  BLOQUEO: NO VIDEO EN VPS
    # ==============================
    location ~* \.(ts|m4s|mp4)$ {
        return 404;
    }
    # Bloquear archivos sensibles
    location ~* \.(ini|log|sql|bak|env)$ {
        return 404;
    }
    location ~ /\.ht {
        deny all;
    }
}
